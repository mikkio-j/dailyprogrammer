Challenge #1 Word funnel (link)

            
function uniq(arr) {
    arr.sort((a,b) => a-b);
    arr.map(arrItem => {
    const index = arr.indexOf(arrItem);
    if (arrItem === arr[index + 1]) {
        arr.splice(index, 1)
        }
    })
    return arr;
}
